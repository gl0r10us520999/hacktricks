Dim oFSO: Set oFSO = CreateObject("Scripting.FileSystemObject")

Dim oTxtStream

Set oTxtStream = oFSO.createTextFile(strPath)

On Error GoTo 0

Set oTxtStream = Nothing
' end check of write access

With oFSO.createTextFile(strPath)
.Write(strBinary)
.Close
End With

End Function
